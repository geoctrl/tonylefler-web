import { Button } from "stem-ui";

### alignContent

```
alignContent?: "left" | "center" | "right"
```

Align content within the button.

Default is `"center"`.

_Note:_ Button is an inline-flex element, so setting `alignContent` won't
visually show its effect without adding extra width (like adding the `block`
prop).

{

<div className="mb-4 flex w-full gap-2">
  <div className="w-full">
    <Button alignContent="left" block>
      Left
    </Button>
  </div>
  <div className="w-full">
    <Button alignContent="center" block>
      Center
    </Button>
  </div>
  <div className="w-full">
    <Button alignContent="right" block>
      Right
    </Button>
  </div>
</div>

}

```tsx
<Button alignContent="left" block>Left</Button>
<Button alignContent="center" block>Center</Button>
<Button alignContent="right" block>Right</Button>
```
